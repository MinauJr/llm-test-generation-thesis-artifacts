// File: src/main/java/example/PasswordPolicy.java
package example;

public final class PasswordPolicy {
    private PasswordPolicy() {
    }

    public static String classify(String password) {
        if (password == null) {
            throw new IllegalArgumentException("password is required");
        }

        if (password.length() < 8) {
            return "weak";
        }

        boolean hasDigit = false;
        boolean hasUppercase = false;
        boolean hasSymbol = false;

        for (int i = 0; i < password.length(); i++) {
            char ch = password.charAt(i);

            if (Character.isDigit(ch)) {
                hasDigit = true;
            } else if (Character.isUpperCase(ch)) {
                hasUppercase = true;
            } else if (!Character.isLetter(ch)) {
                hasSymbol = true;
            }
        }

        int score = 0;

        if (hasDigit) {
            score++;
        }

        if (hasUppercase) {
            score++;
        }

        if (hasSymbol) {
            score++;
        }

        if (score >= 2) {
            return "strong";
        }

        return "medium";
    }
}

// File: src/main/java/example/ShippingCalculator.java
package example;

public final class ShippingCalculator {
    private ShippingCalculator() {
    }

    public static int calculateShippingCents(int quantity, int itemWeightGrams, boolean express, String zone) {
        if (quantity <= 0) {
            throw new IllegalArgumentException("quantity must be positive");
        }

        if (itemWeightGrams <= 0) {
            throw new IllegalArgumentException("weight must be positive");
        }

        if (zone == null) {
            throw new IllegalArgumentException("zone is required");
        }

        String normalisedZone = zone.trim().toLowerCase();

        int baseCost;

        if ("local".equals(normalisedZone)) {
            baseCost = 399;
        } else if ("national".equals(normalisedZone)) {
            baseCost = 699;
        } else if ("international".equals(normalisedZone)) {
            baseCost = 1299;
        } else {
            throw new IllegalArgumentException("unknown zone");
        }

        int totalWeight = quantity * itemWeightGrams;
        int extraCost = 0;

        if (totalWeight > 5000) {
            extraCost = 800;
        } else if (totalWeight > 1000) {
            extraCost = 300;
        }

        if (express) {
            extraCost += 500;
        }

        return baseCost + extraCost;
    }
}
