import pytest
import sut

def test_sqrt_perfect_square():
    result = sut.sqrt(4, 0.0001)
    assert abs(result - 2) < 0.0001

def test_sqrt_non_perfect_square():
    result = sut.sqrt(10, 0.0001)
    assert abs(result - 3.162277660168379) < 0.0001

def test_sqrt_large_number():
    result = sut.sqrt(10000, 0.0001)
    assert abs(result - 100) < 0.0001
