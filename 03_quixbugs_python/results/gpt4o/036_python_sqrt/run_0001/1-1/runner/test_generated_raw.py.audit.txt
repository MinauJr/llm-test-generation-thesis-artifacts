import sut

def test_sqrt_perfect_square():
    result = sut.sqrt(4, 0.0001)
    assert abs(result - 2) < 0.0001

def test_sqrt_non_perfect_square():
    result = sut.sqrt(2, 0.0001)
    assert abs(result - 1.4142) < 0.0001

def test_sqrt_large_number():
    result = sut.sqrt(10000, 0.0001)
    assert abs(result - 100) < 0.0001

def test_sqrt_small_number():
    result = sut.sqrt(0.01, 0.0001)
    assert abs(result - 0.1) < 0.0001

def test_sqrt_epsilon_tolerance():
    result = sut.sqrt(10, 0.1)
    assert abs(result ** 2 - 10) < 0.1
