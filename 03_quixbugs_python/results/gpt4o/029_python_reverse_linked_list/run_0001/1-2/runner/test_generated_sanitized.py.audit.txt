import pytest
import sut

def test_reverse_single_node():
    node = sut.Node()
    result = sut.reverse_linked_list(node)
    assert result == node
    assert result.successor is None

def test_reverse_two_nodes():
    node1 = sut.Node()
    node2 = sut.Node(node1)
    result = sut.reverse_linked_list(node2)
    assert result == node1
    assert result.successor == node2
    assert node2.successor is None

def test_reverse_three_nodes():
    node1 = sut.Node()
    node2 = sut.Node(node1)
    node3 = sut.Node(node2)
    result = sut.reverse_linked_list(node3)
    assert result == node1
    assert result.successor == node2
    assert node2.successor == node3
    assert node3.successor is None

def test_reverse_empty_list():
    result = sut.reverse_linked_list(None)
    assert result is None
