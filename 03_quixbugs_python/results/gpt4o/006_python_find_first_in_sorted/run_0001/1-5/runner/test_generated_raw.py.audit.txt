import sut

def test_find_first_in_sorted_single_element_found():
    assert sut.find_first_in_sorted([5], 5) == 0

def test_find_first_in_sorted_single_element_not_found():
    assert sut.find_first_in_sorted([5], 3) == -1

def test_find_first_in_sorted_multiple_elements_found():
    assert sut.find_first_in_sorted([1, 2, 2, 2, 3], 2) == 1

def test_find_first_in_sorted_multiple_elements_not_found():
    assert sut.find_first_in_sorted([1, 3, 5, 7], 4) == -1

def test_find_first_in_sorted_empty_array():
    assert sut.find_first_in_sorted([], 1) == -1

def test_find_first_in_sorted_first_occurrence():
    assert sut.find_first_in_sorted([1, 1, 1, 2, 3, 3, 4], 1) == 0

def test_find_first_in_sorted_last_element():
    assert sut.find_first_in_sorted([1, 2, 3, 4, 5], 5) == 4

def test_find_first_in_sorted_middle_element():
    assert sut.find_first_in_sorted([1, 2, 3, 4, 5], 3) == 2

def test_find_first_in_sorted_no_match():
    assert sut.find_first_in_sorted([10, 20, 30, 40], 25) == -1

def test_find_first_in_sorted_large_input():
    arr = [1] * 1000 + [2] * 1000 + [3] * 1000
    assert sut.find_first_in_sorted(arr, 2) == 1000
