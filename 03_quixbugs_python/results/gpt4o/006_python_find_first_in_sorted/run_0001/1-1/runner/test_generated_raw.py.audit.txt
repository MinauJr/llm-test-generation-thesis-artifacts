import sut

def test_find_first_in_sorted_single_element_found():
    assert sut.find_first_in_sorted([5], 5) == 0

def test_find_first_in_sorted_single_element_not_found():
    assert sut.find_first_in_sorted([5], 3) == -1

def test_find_first_in_sorted_multiple_elements_found():
    assert sut.find_first_in_sorted([1, 2, 2, 2, 3], 2) == 1

def test_find_first_in_sorted_multiple_elements_not_found():
    assert sut.find_first_in_sorted([1, 2, 3, 4, 5], 6) == -1

def test_find_first_in_sorted_empty_array():
    assert sut.find_first_in_sorted([], 1) == -1
