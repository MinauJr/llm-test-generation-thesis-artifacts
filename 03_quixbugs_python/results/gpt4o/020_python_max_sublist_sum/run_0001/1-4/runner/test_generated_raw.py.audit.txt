import sut

def test_all_positive():
    assert sut.max_sublist_sum([1, 2, 3, 4, 5]) == 15

def test_all_negative():
    assert sut.max_sublist_sum([-1, -2, -3, -4, -5]) == 0

def test_mixed_values():
    assert sut.max_sublist_sum([-2, 1, -3, 4, -1, 2, 1, -5, 4]) == 6

def test_single_element_positive():
    assert sut.max_sublist_sum([10]) == 10

def test_single_element_negative():
    assert sut.max_sublist_sum([-10]) == 0
