import sut

def test_all_positive_numbers():
    assert sut.max_sublist_sum([1, 2, 3, 4, 5]) == 15

def test_all_negative_numbers():
    assert sut.max_sublist_sum([-1, -2, -3, -4, -5]) == 0

def test_mixed_numbers():
    assert sut.max_sublist_sum([-2, 1, -3, 4, -1, 2, 1, -5, 4]) == 6

def test_empty_list():
    assert sut.max_sublist_sum([]) == 0

def test_single_element():
    assert sut.max_sublist_sum([7]) == 7
    assert sut.max_sublist_sum([-7]) == 0
