import sut

def test_kth_single_element():
    assert sut.kth([5], 0) == 5

def test_kth_sorted_array():
    assert sut.kth([1, 2, 3, 4, 5], 2) == 3

def test_kth_reverse_sorted_array():
    assert sut.kth([5, 4, 3, 2, 1], 3) == 4

def test_kth_unsorted_array():
    assert sut.kth([7, 2, 9, 4, 1], 1) == 2

def test_kth_duplicate_elements():
    assert sut.kth([3, 1, 2, 3, 3], 3) == 3
