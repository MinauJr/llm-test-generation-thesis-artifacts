import pytest
import sut

def test_lcs_length_empty_strings():
    assert sut.lcs_length("", "") == 0

def test_lcs_length_no_common_subsequence():
    assert sut.lcs_length("abc", "def") == 0

def test_lcs_length_full_match():
    assert sut.lcs_length("abc", "abc") == 3

@pytest.mark.skip(reason="auto-sanitized failing generated test")
def test_lcs_length_partial_match():
    assert sut.lcs_length("abc", "ac") == 2

def test_lcs_length_single_character_match():
    assert sut.lcs_length("a", "a") == 1
