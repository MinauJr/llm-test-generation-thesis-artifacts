import sut

def test_single_number():
    assert sut.shunting_yard([3]) == [3]

def test_simple_addition():
    assert sut.shunting_yard([3, '+', 4]) == [3, 4, '+']

def test_addition_and_multiplication():
    assert sut.shunting_yard([3, '+', 4, '*', 2]) == [3, 4, 2, '*', '+']

def test_parentheses_precedence():
    assert sut.shunting_yard(['(', 3, '+', 4, ')', '*', 2]) == [3, 4, '+', 2, '*']

def test_multiple_operations():
    assert sut.shunting_yard([3, '+', 4, '*', 2, '-', 1]) == [3, 4, 2, '*', '+', 1, '-']
