import sut

def test_empty_input():
    assert sut.topological_ordering([]) == []

def test_single_node():
    node = sut.Node()
    assert sut.topological_ordering([node]) == [node]

def test_linear_dependency():
    node1 = sut.Node()
    node2 = sut.Node(incoming_nodes=[node1])
    node3 = sut.Node(incoming_nodes=[node2])
    assert sut.topological_ordering([node3, node2, node1]) == [node1, node2, node3]

def test_circular_dependency():
    node1 = sut.Node()
    node2 = sut.Node(incoming_nodes=[node1])
    node1.incoming_nodes = [node2]
    assert sut.topological_ordering([node1, node2]) == []

def test_multiple_independent_nodes():
    node1 = sut.Node()
    node2 = sut.Node()
    node3 = sut.Node()
    result = sut.topological_ordering([node1, node2, node3])
    assert set(result) == {node1, node2, node3} and len(result) == 3
