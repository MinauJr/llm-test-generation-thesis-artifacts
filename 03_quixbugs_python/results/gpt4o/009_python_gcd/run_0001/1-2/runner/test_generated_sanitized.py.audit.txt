import pytest
import sut

def test_gcd_positive_numbers():
    assert sut.gcd(48, 18) == 6

def test_gcd_one_zero():
    assert sut.gcd(0, 5) == 5
    assert sut.gcd(7, 0) == 7

def test_gcd_both_zeros():
    assert sut.gcd(0, 0) == 0

def test_gcd_negative_numbers():
    assert sut.gcd(-48, -18) == -6

@pytest.mark.skip(reason="auto-sanitized failing generated test")
def test_gcd_mixed_signs():
    assert sut.gcd(-48, 18) == 6
    assert sut.gcd(48, -18) == 6
