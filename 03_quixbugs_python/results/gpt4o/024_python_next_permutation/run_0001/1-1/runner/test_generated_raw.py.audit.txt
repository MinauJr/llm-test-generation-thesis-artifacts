import sut

def test_next_permutation_middle_swap():
    assert sut.next_permutation([1, 2, 3]) == [1, 3, 2]

def test_next_permutation_last_swap():
    assert sut.next_permutation([1, 3, 2]) == [2, 1, 3]

def test_next_permutation_reverse_to_start():
    assert sut.next_permutation([3, 2, 1]) == None

def test_next_permutation_single_element():
    assert sut.next_permutation([1]) == None

def test_next_permutation_no_change():
    assert sut.next_permutation([1, 1, 1]) == None
