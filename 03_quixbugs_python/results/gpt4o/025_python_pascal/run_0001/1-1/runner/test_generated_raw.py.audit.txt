import sut

def test_pascal_1():
    assert sut.pascal(1) == [[1]]

def test_pascal_2():
    assert sut.pascal(2) == [[1], [1, 1]]

def test_pascal_3():
    assert sut.pascal(3) == [[1], [1, 1], [1, 2, 1]]

def test_pascal_4():
    assert sut.pascal(4) == [[1], [1, 1], [1, 2, 1], [1, 3, 3, 1]]

def test_pascal_5():
    assert sut.pascal(5) == [[1], [1, 1], [1, 2, 1], [1, 3, 3, 1], [1, 4, 6, 4, 1]]
