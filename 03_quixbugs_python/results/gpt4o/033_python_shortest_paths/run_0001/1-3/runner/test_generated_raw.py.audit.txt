import sut

def test_single_edge():
    weight_by_edge = {('A', 'B'): 5}
    result = sut.shortest_paths('A', weight_by_edge)
    assert result == {'A': 0, 'B': 5}

def test_two_edges_linear():
    weight_by_edge = {('A', 'B'): 3, ('B', 'C'): 4}
    result = sut.shortest_paths('A', weight_by_edge)
    assert result == {'A': 0, 'B': 3, 'C': 7}

def test_two_edges_branching():
    weight_by_edge = {('A', 'B'): 2, ('A', 'C'): 5}
    result = sut.shortest_paths('A', weight_by_edge)
    assert result == {'A': 0, 'B': 2, 'C': 5}

def test_cycle():
    weight_by_edge = {('A', 'B'): 1, ('B', 'C'): 2, ('C', 'A'): 3}
    result = sut.shortest_paths('A', weight_by_edge)
    assert result == {'A': 0, 'B': 1, 'C': 3}

def test_disconnected_graph():
    weight_by_edge = {('A', 'B'): 4, ('C', 'D'): 6}
    result = sut.shortest_paths('A', weight_by_edge)
    assert result == {'A': 0, 'B': 4, 'C': float('inf'), 'D': float('inf')}
