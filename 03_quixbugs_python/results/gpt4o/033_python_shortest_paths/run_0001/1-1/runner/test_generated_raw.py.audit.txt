import sut

def test_single_edge():
    source = 'A'
    weight_by_edge = {('A', 'B'): 5}
    result = sut.shortest_paths(source, weight_by_edge)
    assert result == {'A': 0, 'B': 5}

def test_two_edges():
    source = 'A'
    weight_by_edge = {('A', 'B'): 5, ('B', 'C'): 3}
    result = sut.shortest_paths(source, weight_by_edge)
    assert result == {'A': 0, 'B': 5, 'C': 8}

def test_no_edges():
    source = 'A'
    weight_by_edge = {}
    result = sut.shortest_paths(source, weight_by_edge)
    assert result == {'A': 0}

def test_multiple_paths():
    source = 'A'
    weight_by_edge = {('A', 'B'): 5, ('A', 'C'): 2, ('B', 'C'): 1}
    result = sut.shortest_paths(source, weight_by_edge)
    assert result == {'A': 0, 'B': 5, 'C': 2}

def test_negative_weight():
    source = 'A'
    weight_by_edge = {('A', 'B'): 4, ('B', 'C'): -2, ('A', 'C'): 5}
    result = sut.shortest_paths(source, weight_by_edge)
    assert result == {'A': 0, 'B': 4, 'C': 2}
